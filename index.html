function buscarSeccion() { const seccion = document .getElementById("seccionInput") .value .trim(); if (!seccion) return; const features = map.querySourceFeatures( "secciones-local" ); const encontrada = features.find( f => String(f.properties.SECCION_2026) === seccion ); if (!encontrada) { alert("Sección no encontrada"); return; } map.setFilter( "resaltado-seccion", ["==", "SECCION_2026", encontrada.properties.SECCION_2026] ); const bounds = new mapboxgl.LngLatBounds(); encontrada.geometry.coordinates[0].forEach(coord => { bounds.extend(coord); }); map.fitBounds(bounds, { padding: 40 }); }
